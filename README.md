## Usage
```
$ npm i ot-hapi-health --save
```

```
var server = new (require('hapi').Server)();
server.connection({ port: 3000 });

server.register([
    require('ot-hapi-health')
], function(err) {
    if (err) {
        console.error('Failed to load plugin:', err);
    }

    server.start();
});
```
